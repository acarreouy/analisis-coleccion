function showSection(id) {
  document.querySelectorAll('.section').forEach(s => s.classList.remove('active'));
  document.querySelectorAll('.nav-item').forEach(n => n.classList.remove('active'));
  document.getElementById(id).classList.add('active');
  const sectionMap = {s0:0,s1:1,s2:2,s3:3,s4:4,s5:5,s6:6,s7:7,s8:8,s9:9,s10:10};
  document.querySelectorAll('.nav-item')[sectionMap[id]].classList.add('active');
  setTimeout(() => renderCharts(), 50);
}

        labels:DATA.renovProm.map(d=>d.t.substring(0,28)+'…'),
        datasets:[{data:DATA.renovProm.map(d=>d.avg), backgroundColor:DATA.renovProm.map(d=>d.avg>6?COLORS.red:d.avg>4?COLORS.orange:COLORS.yellow), borderRadius:4}]
      },
      options: baseOptions({ indexAxis:'y', scales:{x:{grid:gridStyle,ticks:tickStyle},y:{grid:gridStyle,ticks:{...tickStyle,font:{size:9}}}} })
    });

    // Table
    const tb9 = document.getElementById('tb_renov');
    DATA.renovProm.forEach((d,i) => {
      tb9.innerHTML += `<tr>
        <td><span class="rank-num rank-${i<3?i+1:''}">${i+1}</span></td>
        <td style="max-width:280px;font-size:0.76rem">${d.t}</td>
        <td>${d.p}</td>
        <td style="color:${d.avg>5?'var(--red)':d.avg>3?'var(--orange)':'var(--text)'}"><strong>${d.avg}</strong></td>
        <td style="font-size:0.72rem">${d.sem}</td>
      </tr>`;
    });

    new Chart(document.getElementById('c_sugeridos'), {
      type:'bar',
      data:{
        labels:DATA.sugeridos.map(d=>d.t.substring(0,25)+'…'),
        datasets:[
          {label:'Préstamos', data:DATA.sugeridos.map(d=>d.p), backgroundColor:COLORS.blue, borderRadius:4},
          {label:'Prom. Renovaciones', data:DATA.sugeridos.map(d=>d.avg), backgroundColor:COLORS.orange, borderRadius:4, yAxisID:'y2'}
        ]
      },
      options: baseOptions({
        plugins:{legend:{display:true,labels:{color:'#bfbfbf',boxWidth:12}},tooltip:baseOptions().plugins.tooltip},
        scales:{
          x:{grid:gridStyle,ticks:{...tickStyle,font:{size:9}}},
          y:{grid:gridStyle,ticks:tickStyle,position:'left'},
          y2:{grid:{display:false},ticks:{...tickStyle,callback:v=>v+'x'},position:'right'}
        }
      })
    });
  }

  // ── S10 ──
  if (!chartsCreated.s10) {
    chartsCreated.s10 = true;

    // Tables
    const tbUL = document.getElementById('tb_users_lib');
    DATA.usuarios_libros.forEach((d,i) => {
      tbUL.innerHTML += `<tr>
        <td><span class="rank-num rank-${i<3?i+1:''}">${i+1}</span></td>
        <td>${d.u}</td>
        <td><strong style="color:var(--text)">${d.p}</strong></td>
        <td>${d.d}</td>
        <td style="font-size:0.72rem">${d.sem}</td>
      </tr>`;
    });

    const tbUE = document.getElementById('tb_users_esp');
    DATA.usuarios_esp.forEach((d,i) => {
      tbUE.innerHTML += `<tr>
        <td><span class="rank-num rank-${i<3?i+1:''}">${i+1}</span></td>
        <td>${d.u}</td>
        <td><strong style="color:${d.u==='USER_1284'?'var(--red)':'var(--text)}'}">${d.p}</strong></td>
        <td>${d.d}</td>
        <td style="font-size:0.72rem">${d.sem}</td>
      </tr>`;
    });

    // Ubicaciones
    new Chart(document.getElementById('c_ubicaciones'), {
      type:'bar',
      data:{
        labels:DATA.ubicaciones.map(d=>'Sig.'+d.sig),
        datasets:[{data:DATA.ubicaciones.map(d=>d.p), backgroundColor:PALETTE, borderRadius:4}]
      },
      options: baseOptions({ indexAxis:'y', scales:{x:{grid:gridStyle,ticks:tickStyle},y:{grid:gridStyle,ticks:tickStyle}} })
    });

    // Heatmap
    const hm = DATA.heatmap;
    const hmEl = document.getElementById('heatmap_container');

    const maxVal = Math.max(...hm.data.flat());

    let html = `<div style="display:grid;grid-template-columns:160px repeat(${hm.cols.length},1fr);gap:3px;">`;

    // Header row
    html += `<div></div>`;
    hm.cols.forEach(c => {
      html += `<div style="text-align:center;font-size:0.7rem;color:var(--text-sec);padding:4px;font-weight:700">${c}</div>`;
    });

    // Data rows
    hm.data.forEach((row, ri) => {
      html += `<div style="font-size:0.72rem;color:var(--text-sec);padding:4px 0;display:flex;align-items:center">${hm.rows[ri]}</div>`;
      row.forEach(val => {
        const pct = val / maxVal;
        const alpha = Math.max(0.1, pct);
        const r = ri < 2 ? '76,120,168' : ri < 4 ? '118,183,178' : ri === 4 ? '176,122,161' : '89,161,79';
        html += `<div class="heat-cell" title="${val}" style="background:rgba(${r},${alpha});height:38px;font-size:${val>0?'0.7rem':'0'}">${val>0?val.toLocaleString():''}</div>`;
      });
    });

    html += '</div>';

    // Legend
    html += `<div style="margin-top:12px;display:flex;align-items:center;gap:8px;font-size:0.7rem;color:var(--text-sec);">
      <span>Baja</span>
      <div style="display:flex;gap:2px">`;
    for(let i=0;i<=10;i++){
      html += `<div style="width:18px;height:10px;border-radius:2px;background:rgba(76,120,168,${i/10})"></div>`;
    }
    html += `</div><span>Alta</span></div>`;

    hmEl.innerHTML = html;
  }
}
